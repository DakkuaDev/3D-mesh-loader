130  void View::update (float delta)
131  {
132 
133  for (unsigned i = 0; i < total_meshes; i++)
134  {
135 
136  // 1. Se actualizan los parámetros de transformatión (sólo se modifica el ángulo):
137 
138  static float angle = 0;
139  angle += delta;
140 
141  // 2. Se trasforma la entidad
142  entities.at(i).scale_entity(0.05f);
143  entities.at(i).rotate_entity_y(angle);
144  entities.at(i).traslate_entity(0.f, 3.f, -3.f);
145 
146 
147  // Creación de la matriz de transformación unificada:
148  //Matrix44 projection = perspective(20, 1, 15, float(width) / height);
149  Matrix44 projection = camera->get_projection();
150  Matrix44 transformation = camera->get_camera_matrix() * entities.at(i).update_trasform(projection);
151 
152 
153  // 3. Iluminación de la escena
154 
155  if (bake_scene)
156  {
157  Light light(Vector3f(5.f, 10.f, 15.f));
158  //Vector4f directional_light = light.get_ambient_light();
159  //float vision_angle = 30;
160 
161 
162  for (size_t index = 0, number_of_vertices = original_vertices_vector[i].size(); index < number_of_vertices; index++)
163  {
164  // Calculo la luz (Ambiental + Difusa)
165  Vector4f directional_light = Vector4f(light.calculate_light(original_normals_vector[i][index]), 1);
166  float intensity = (directional_light.x, directional_light.y, directional_light.z);
167 
168  if (intensity > 1.0f) { intensity = 1.0f; }
169  else if (intensity < 0.0f) { intensity = 0.0f; }
170 
171  // Se la aplico al buffer de colores
172  original_colors_vector[i][index].set(intensity, intensity, intensity);
173 
174  }
175  }
176 
177 
178  // 4. Se transforman todos los vértices usando la matriz de transformación resultante:
179 
180  for (size_t index = 0, number_of_vertices = original_vertices_vector[i].size(); index < number_of_vertices; index++)
181  {
182  // Se multiplican todos los vértices originales con la matriz de transformación y
183  // se guarda el resultado en otro vertex buffer:
184  Vertex & vertex = transformed_vertices_vector[i][index] = transformation * original_vertices_vector[i][index];
185 
186  // La matriz de proyección en perspectiva hace que el último componente del vector
187  // transformado no tenga valor 1.0, por lo que hay que normalizarlo dividiendo:
188 
189  float divisor = 1.f / vertex.w;
190 
191  vertex.x *= divisor;
192  vertex.y *= divisor;
193  vertex.z *= divisor;
194  vertex.w = 1.f;
195  }
196  }

199  void View::render ()
200  {
201  // Se convierten las coordenadas transformadas y proyectadas a coordenadas
202  // de recorte (-1 a +1) en coordenadas de pantalla con el origen centrado.
203  // La coordenada Z se escala a un valor suficientemente grande dentro del
204  // rango de int (que es lo que espera fill_convex_polygon_z_buffer).
205 
206  rasterizer.clear ();
207 
208  for (unsigned i = 0; i < total_meshes; i++)
209  {
210 
211  Matrix44 identity(1);
212  Matrix44 scaling = scale (identity, float(width / 2), float(height / 2), 100000000.f);
213  Matrix44 translation = translate (identity, Vector3f{ float(width / 2), float(height / 2), 0.f });
214  Matrix44 transformation = translation * scaling;
215 
216  for (size_t index = 0, number_of_vertices = transformed_vertices_vector[i].size (); index < number_of_vertices; index++)
217  {
218  display_vertices_vector[i][index] = Point4i( transformation * transformed_vertices_vector[i][index] );
219  }
220 
221  // Se borra el framebúffer y se dibujan los triángulos:
222 
223 
224  for (int * indices = original_indices_vector[i].data (), * end = indices + original_indices_vector[i].size (); indices < end; indices += 3)
225  {
226  if (is_frontface (transformed_vertices_vector[i].data (), indices))
227  {
228  // Se establece el color del polígono a partir del color de su primer vértice:
229 
230  rasterizer.set_color (original_colors_vector[i][*indices]);
231 
232  // Vértices que se salen del viewport (recorte geométrico)
233  int clipped_vertex_count = clip_polygon->clipping_polygon();
234 
235  // Se rellena el polígono:
236 
237  rasterizer.fill_convex_polygon_z_buffer (display_vertices_vector[i].data (), indices, indices + 3);
238  }
239  }
240 
241  // Se copia el framebúffer oculto en el framebúffer de la ventana:
242 
243  color_buffer.blit_to_window ();
244 
245  }

248  bool View::is_frontface (const Vertex * const projected_vertices, const int * const indices)
249  {
250  const Vertex & v0 = projected_vertices[indices[0]];
251  const Vertex & v1 = projected_vertices[indices[1]];
252  const Vertex & v2 = projected_vertices[indices[2]];
253 
254  // Se asumen coordenadas proyectadas y polígonos definidos en sentido horario.
255  // Se comprueba a qué lado de la línea que pasa por v0 y v1 queda el punto v2:
256 
257  return ((v1[0] - v0[0]) * (v2[1] - v0[1]) - (v2[0] - v0[0]) * (v1[1] - v0[1]) < 0.f);
258  }
